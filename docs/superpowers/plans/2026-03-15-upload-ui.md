# DOC-12: Invoice Upload UI Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build the drag-and-drop invoice upload page — the user's first real interaction with Docket.

**Architecture:** Single `UploadZone` client component handles all state (idle, dragging, uploading, success), validation, drag events, and mock upload. The upload page is a thin wrapper with a heading. No external libraries — native HTML5 drag events + Tailwind only.

**Tech Stack:** Next.js 14 (App Router), React, TypeScript, Tailwind CSS, Vitest + Testing Library

**Spec:** `docs/superpowers/specs/2026-03-15-upload-ui-design.md`

import { describe, it, expect } from 'vitest'
import { render, screen } from '@testing-library/react'
import userEvent from '@testing-library/user-event'
import UploadZone from './UploadZone'

function createFile(name: string, size: number, type: string): File {
  const buffer = new ArrayBuffer(size)
  return new File([buffer], name, { type })
}

describe('UploadZone', () => {
  it('renders idle state by default', () => {
    render(<UploadZone />)
    expect(screen.getByText(/drag & drop your invoice/i)).toBeInTheDocument()
    expect(screen.getByText(/or click to browse/i)).toBeInTheDocument()
  })

  describe('validation', () => {
    it('rejects files over 10MB', async () => {
      const user = userEvent.setup()
      render(<UploadZone />)
      const input = document.querySelector('input[type="file"]') as HTMLInputElement
      const bigFile = createFile('huge.pdf', 11 * 1024 * 1024, 'application/pdf')
      await user.upload(input, bigFile)
      expect(screen.getByText(/file exceeds 10mb limit/i)).toBeInTheDocument()

    it('rejects unsupported file types', async () => {
      const user = userEvent.setup()

      const user = userEvent.setup()

## Task 5: Implement UploadZone component

**Files:**
- Rewrite: `components/invoices/UploadZone.tsx`

- [ ] **Step 1: Implement the full UploadZone component**

```tsx
'use client'

import { useState, useRef, useCallback } from 'react'

type UploadState = 'idle' | 'dragging' | 'uploading' | 'success'

const ACCEPTED_TYPES: Record<string, string> = {
  'application/pdf': '.pdf',
  'image/jpeg': '.jpg',
  'image/png': '.png',
}
const MAX_FILE_SIZE = 10 * 1024 * 1024 // 10MB

function validateFile(file: File): string | null {
  if (!ACCEPTED_TYPES[file.type]) {
    return 'Unsupported file type. Please upload a PDF, JPG, or PNG.'
  }
  if (file.size > MAX_FILE_SIZE) {
    return 'File exceeds 10MB limit.'
  }
  return null
}

export default function UploadZone() {
  const [state, setState] = useState<UploadState>('idle')
  const [error, setError] = useState<string | null>(null)
  const [progress, setProgress] = useState(0)
  const [fileName, setFileName] = useState<string | null>(null)
  const [statusAnnouncement, setStatusAnnouncement] = useState('')
  const inputRef = useRef<HTMLInputElement>(null)
  const dragCounter = useRef(0)

  const handleFile = useCallback((file: File) => {
    setError(null)

    const validationError = validateFile(file)
    if (validationError) {
      setError(validationError)
      return
    }

    setFileName(file.name)
    setState('uploading')
    setProgress(0)
    setStatusAnnouncement(`Uploading ${file.name}`)

    // Mock upload — replace with real fetch to /api/invoices/upload in EXT-2
    const steps = [30, 60, 90, 100]
    steps.forEach((pct, i) => {
      setTimeout(() => {
        setProgress(pct)
        if (pct === 100) {
          setState('success')
          setStatusAnnouncement('Upload complete')
        }
      }, (i + 1) * 400)
    })
  }, [])

  const handleFiles = useCallback((files: FileList | null) => {
    if (!files || files.length === 0) return
    if (state === 'uploading') return

    if (files.length > 1) {
      setError('Please upload one file at a time.')
      return
    }

    handleFile(files[0])
  }, [state, handleFile])

  const handleDragEnter = useCallback((e: React.DragEvent) => {
    e.preventDefault()
    e.stopPropagation()
    dragCounter.current++
    if (state !== 'uploading') {
      setState('dragging')
    }
  }, [state])

  const handleDragOver = useCallback((e: React.DragEvent) => {
    e.preventDefault()
    e.stopPropagation()
  }, [])

  const handleDragLeave = useCallback((e: React.DragEvent) => {
    e.preventDefault()
    e.stopPropagation()
    dragCounter.current--
    if (dragCounter.current === 0 && state === 'dragging') {
      setState('idle')
    }
  }, [state])

  const handleDrop = useCallback((e: React.DragEvent) => {
    e.preventDefault()
    e.stopPropagation()
    dragCounter.current = 0
    if (state === 'uploading') return
    setState('idle')
    handleFiles(e.dataTransfer.files)
  }, [state, handleFiles])

  const handleClick = useCallback(() => {
    if (state === 'uploading') return
    inputRef.current?.click()
  }, [state])

  const handleKeyDown = useCallback((e: React.KeyboardEvent) => {
    if (e.key === 'Enter' || e.key === ' ') {
      e.preventDefault()
      handleClick()
    }
  }, [handleClick])

  const handleInputChange = useCallback((e: React.ChangeEvent<HTMLInputElement>) => {
    handleFiles(e.target.files)
    // Reset input value so the same file can be re-selected
    if (inputRef.current) inputRef.current.value = ''
  }, [handleFiles])

  const handleReset = useCallback(() => {
    setState('idle')
    setError(null)
    setProgress(0)
    setFileName(null)
    setStatusAnnouncement('')
  }, [])

  return (
    <div>
      <div
        role="button"
        tabIndex={0}
        aria-label="Upload invoice file"
        aria-describedby={error ? 'upload-error' : undefined}
        onClick={handleClick}
        onKeyDown={handleKeyDown}
        onDragEnter={handleDragEnter}
        onDragOver={handleDragOver}
        onDragLeave={handleDragLeave}
        onDrop={handleDrop}
        className={`
          relative flex min-h-[200px] items-center justify-center rounded-lg border-2 border-dashed
          transition-colors duration-200 md:min-h-[300px]
          ${state === 'dragging'
            ? 'border-accent bg-blue-50'
            : state === 'uploading'
              ? 'cursor-default border-gray-300 bg-white'
              : 'cursor-pointer border-gray-300 bg-white hover:border-gray-400'
          }
          focus:outline-none focus:ring-2 focus:ring-accent focus:ring-offset-2
          p-4 md:p-8
        `}
      >
        <input
          ref={inputRef}
          type="file"
          accept=".pdf,.jpg,.jpeg,.png"
          onChange={handleInputChange}
          className="hidden"
        />

        {state === 'idle' && (
          <div className="flex flex-col items-center text-center">
            {/* Cloud upload icon */}
            <svg
              className="mb-4 h-12 w-12 text-gray-400"
              fill="none"
              viewBox="0 0 24 24"
              stroke="currentColor"
              strokeWidth={1.5}
            >
              <path
                strokeLinecap="round"
                strokeLinejoin="round"
                d="M12 16.5V9.75m0 0 3 3m-3-3-3 3M6.75 19.5a4.5 4.5 0 0 1-1.41-8.775 5.25 5.25 0 0 1 10.233-2.33 3 3 0 0 1 3.758 3.848A3.752 3.752 0 0 1 18 19.5H6.75Z"
              />
            </svg>
            <p className="text-base font-medium text-primary">
              Drag & drop your invoice
            </p>
            <p className="mt-1 text-sm text-gray-500">
              or click to browse
            </p>
            <p className="mt-3 text-xs text-gray-400">
              PDF, JPG, or PNG — up to 10MB
            </p>
          </div>
        )}

        {state === 'dragging' && (
          <div className="flex flex-col items-center text-center">
            <svg
              className="mb-4 h-12 w-12 text-accent"
              fill="none"
              viewBox="0 0 24 24"
              stroke="currentColor"
              strokeWidth={1.5}
            >
              <path
                strokeLinecap="round"
                strokeLinejoin="round"
                d="M12 16.5V9.75m0 0 3 3m-3-3-3 3M6.75 19.5a4.5 4.5 0 0 1-1.41-8.775 5.25 5.25 0 0 1 10.233-2.33 3 3 0 0 1 3.758 3.848A3.752 3.752 0 0 1 18 19.5H6.75Z"
              />
            </svg>
            <p className="text-base font-medium text-accent">
              Drop your file here
            </p>
          </div>
        )}

        {state === 'uploading' && (
          <div className="flex w-full max-w-xs flex-col items-center text-center">
            <p className="mb-3 truncate text-sm font-medium text-primary" title={fileName ?? ''}>
              {fileName}
            </p>
            <div
              className="h-2 w-full overflow-hidden rounded-full bg-gray-200"
              role="progressbar"
              aria-valuenow={progress}
              aria-valuemin={0}
              aria-valuemax={100}
            >
              <div
                className="h-full rounded-full bg-accent transition-all duration-300 ease-out"
                style={{ width: `${progress}%` }}
              />
            </div>
            <p className="mt-2 text-sm text-gray-500">
              Uploading...
            </p>
          </div>
        )}

        {state === 'success' && (
          <div className="flex flex-col items-center text-center">
            {/* Green checkmark icon */}
            <svg
              className="mb-3 h-10 w-10 text-success"
              fill="none"
              viewBox="0 0 24 24"
              stroke="currentColor"
              strokeWidth={2}
            >
              <path
                strokeLinecap="round"
                strokeLinejoin="round"
                d="M9 12.75 11.25 15 15 9.75M21 12a9 9 0 1 1-18 0 9 9 0 0 1 18 0Z"
              />
            </svg>
            <p className="text-sm font-medium text-primary">
              {fileName}
            </p>
            <div className="mt-2 flex items-center gap-1.5 text-sm text-gray-500">
              <span className="inline-block h-2 w-2 animate-pulse rounded-full bg-amber-400" />
              Processing...
            </div>
            <button
              type="button"
              onClick={(e) => {
                e.stopPropagation()
                handleReset()
              }}
              className="mt-4 rounded-md border border-gray-300 px-4 py-2 text-sm font-medium text-gray-700 hover:bg-gray-50"
            >
              Upload Another
            </button>
          </div>
        )}
      </div>

      {/* Error message */}
      {error && (
        <p id="upload-error" className="mt-2 text-sm text-error">
          {error}
        </p>
      )}

      {/* Screen reader announcements */}
      <div aria-live="polite" className="sr-only">
        {statusAnnouncement}
      </div>
    </div>
  )
}
```
